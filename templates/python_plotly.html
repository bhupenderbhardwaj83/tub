Getting Started with Plotly

Installation & Setup
WHAT IT DOES

Plotly creates interactive, web-based visualizations. Works in Jupyter notebooks, web apps, and exports to HTML.

EXAMPLE 
# Install
# pip install plotly

import plotly.express as px
import plotly.graph_objects as go

# Simple example
fig = px.line(x=[1, 2, 3], y=[2, 4, 6], title='Simple Line')
fig.show()

# Save as HTML
fig.write_html('chart.html')
Plotly Express vs Graph Objects
WHAT IT DOES

Plotly Express (px) - Simple, concise API for quick charts
Graph Objects (go) - Full control, more customization

Line Charts

Basic Line Chart
EXAMPLE 
import plotly.express as px
import pandas as pd

# Sample data
df = pd.DataFrame({
    'year': [2020, 2021, 2022, 2023, 2024],
    'revenue': [100, 150, 200, 280, 350]
})

# Create line chart
fig = px.line(
    df,
    x='year',
    y='revenue',
    title='Revenue Growth',
    labels={'revenue': 'Revenue ($M)'},
    markers=True
)

fig.show()
Multiple Lines
EXAMPLE 
import plotly.express as px

df = pd.DataFrame({
    'month': ['Jan', 'Feb', 'Mar', 'Apr'],
    'Product A': [100, 120, 140, 160],
    'Product B': [90, 110, 130, 145]
})

# Melt dataframe for multiple lines
df_melted = df.melt(id_vars='month', var_name='Product', value_name='Sales')

fig = px.line(
    df_melted,
    x='month',
    y='Sales',
    color='Product',  # Creates separate line per product
    title='Sales Comparison'
)

fig.show()

Bar Charts

Basic Bar Chart
EXAMPLE 
import plotly.express as px

df = pd.DataFrame({
    'category': ['A', 'B', 'C', 'D'],
    'value': [25, 40, 30, 50]
})

fig = px.bar(
    df,
    x='category',
    y='value',
    title='Sales by Category',
    text='value',  # Show values on bars
    color='value',  # Color gradient
    color_continuous_scale='Blues'
)

fig.show()
Grouped & Stacked Bars
EXAMPLE 
import plotly.express as px

df = pd.DataFrame({
    'region': ['North', 'North', 'South', 'South'],
    'product': ['A', 'B', 'A', 'B'],
    'sales': [100, 150, 120, 180]
})

# Grouped bars
fig = px.bar(df, x='region', y='sales', color='product',
             barmode='group', title='Grouped Bars')

# Stacked bars
fig = px.bar(df, x='region', y='sales', color='product',
             barmode='stack', title='Stacked Bars')

fig.show()
Horizontal Bar Chart
EXAMPLE 
fig = px.bar(
    df,
    y='category',  # Swap x and y
    x='value',
    orientation='h',  # Horizontal
    title='Horizontal Bars'
)

fig.show()

Scatter Plots

Basic Scatter Plot
EXAMPLE 
import plotly.express as px

df = pd.DataFrame({
    'x': [1, 2, 3, 4, 5],
    'y': [2, 5, 4, 8, 7],
    'size': [20, 40, 30, 50, 45],
    'color': ['A', 'B', 'A', 'B', 'A']
})

fig = px.scatter(
    df,
    x='x',
    y='y',
    size='size',  # Point size
    color='color',  # Point color by category
    title='Scatter Plot',
    hover_data=['x', 'y', 'size']
)

fig.show()
Scatter with Trendline
EXAMPLE 
import plotly.express as px

df = px.data.tips()  # Built-in dataset

fig = px.scatter(
    df,
    x='total_bill',
    y='tip',
    trendline='ols',  # Add trendline
    title='Tips vs Bill with Trendline'
)

fig.show()
Real-World Example: Sales Dashboard
import plotly.graph_objects as go
from plotly.subplots import make_subplots

# Create subplots
fig = make_subplots(
    rows=2, cols=2,
    subplot_titles=('Revenue', 'Users', 'Growth', 'Distribution')
)

# Add line chart
fig.add_trace(
    go.Scatter(x=[1, 2, 3], y=[100, 150, 200], name='Revenue'),
    row=1, col=1
)

# Add bar chart
fig.add_trace(
    go.Bar(x=[1, 2, 3], y=[50, 75, 100], name='Users'),
    row=1, col=2
)

fig.update_layout(height=600, title_text='Sales Dashboard')
fig.show()

Advanced Features

Customization & Styling
EXAMPLE 
import plotly.express as px

fig = px.line(x=[1, 2, 3], y=[2, 4, 6])

# Update layout
fig.update_layout(
    title='Custom Styled Chart',
    title_font_size=24,
    xaxis_title='X Axis',
    yaxis_title='Y Axis',
    template='plotly_dark',  # Dark theme
    showlegend=True,
    hovermode='x unified'
)

# Update traces
fig.update_traces(
    line_color='#FF6B6B',
    line_width=3,
    marker=dict(size=10)
)

fig.show()
Interactive Features
EXAMPLE 
import plotly.express as px

df = px.data.gapminder()

# Animation frame
fig = px.scatter(
    df,
    x='gdpPercap',
    y='lifeExp',
    size='pop',
    color='continent',
    hover_name='country',
    animation_frame='year',  # Animate by year
    animation_group='country',
    log_x=True,
    range_y=[25, 90]
)

fig.show()
Export & Save
EXAMPLE 
import plotly.express as px

fig = px.line(x=[1, 2, 3], y=[2, 4, 6])

# Save as HTML
fig.write_html('chart.html')

# Save as static image (requires kaleido)
# pip install -U kaleido
fig.write_image('chart.png')
fig.write_image('chart.pdf')

# Show in browser
fig.show()
More Chart Types
EXAMPLE 
import plotly.express as px

# Pie chart
fig = px.pie(values=[30, 20, 50], names=['A', 'B', 'C'])

# Histogram
fig = px.histogram(df, x='value', nbins=20)

# Box plot
fig = px.box(df, x='category', y='value')

# Heatmap
fig = px.imshow([[1, 2, 3], [4, 5, 6]])

# Sunburst chart
fig = px.sunburst(df, path=['region', 'category'], values='sales')

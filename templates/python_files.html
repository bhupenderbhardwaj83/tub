Basic File Operations

Reading Files
WHAT IT DOES

Open and read file contents using open() function with 'r' mode (read).

EXAMPLE 
# Read entire file
file = open('data.txt', 'r')
content = file.read()
print(content)
file.close()

# Read line by line
file = open('data.txt', 'r')
for line in file:
    print(line.strip())
file.close()

# Read all lines into list
file = open('data.txt', 'r')
lines = file.readlines()
file.close()
Writing Files
EXAMPLE 
# Write (overwrites existing file)
file = open('output.txt', 'w')
file.write("Hello, World!
")
file.write("Second line
")
file.close()

# Append (adds to existing file)
file = open('output.txt', 'a')
file.write("Appended line
")
file.close()

# Write multiple lines
lines = ["Line 1
", "Line 2
", "Line 3
"]
file = open('output.txt', 'w')
file.writelines(lines)
file.close()
File Modes
WHAT IT DOES

Different modes for opening files: 'r' (read), 'w' (write), 'a' (append), 'r+' (read+write), 'b' (binary). 

# Common modes
'r'   # Read only (default)
'w'   # Write (creates new file or truncates existing)
'a'   # Append (adds to end of file)
'r+'  # Read and write
'rb'  # Read binary
'wb'  # Write binary

Context Managers (with statement)

Using 'with' Statement
WHAT IT DOES

The 'with' statement automatically closes files, even if an error occurs. Best practice for file handling.

EXAMPLE 
# Automatically closes file
with open('data.txt', 'r') as file:
    content = file.read()
    print(content)
# File automatically closed here

# Write with context manager
with open('output.txt', 'w') as file:
    file.write("Hello, World!")

# Multiple files at once
with open('input.txt') as infile, open('output.txt', 'w') as outfile:
    content = infile.read()
    outfile.write(content.upper())
Real-World Example: Log File Processing
def filter_errors(input_file, output_file):
    with open(input_file) as infile, open(output_file, 'w') as outfile:
        for line in infile:
            if 'ERROR' in line:
                outfile.write(line)

filter_errors('app.log', 'errors.log')

Working with CSV Files

Reading CSV Files
EXAMPLE 
import csv

# Read CSV file
with open('data.csv', 'r') as file:
    reader = csv.reader(file)
    for row in reader:
        print(row)  # Each row is a list

# Read as dictionary
with open('data.csv') as file:
    reader = csv.DictReader(file)
    for row in reader:
        print(row['name'], row['age'])  # Access by column name
Writing CSV Files
EXAMPLE 
import csv

# Write CSV file
data = [
    ['Name', 'Age', 'City'],
    ['Alice', 25, 'NYC'],
    ['Bob', 30, 'LA']
]

with open('output.csv', 'w', newline='') as file:
    writer = csv.writer(file)
    writer.writerows(data)

# Write dictionary to CSV
data = [
    {'name': 'Alice', 'age': 25, 'city': 'NYC'},
    {'name': 'Bob', 'age': 30, 'city': 'LA'}
]

with open('output.csv', 'w', newline='') as file:
    fieldnames = ['name', 'age', 'city']
    writer = csv.DictWriter(file, fieldnames=fieldnames)
    writer.writeheader()
    writer.writerows(data)

Working with JSON Files

Reading & Writing JSON
EXAMPLE 
import json

# Read JSON file
with open('data.json') as file:
    data = json.load(file)
    print(data)

# Write JSON file
data = {
    'name': 'Alice',
    'age': 25,
    'hobbies': ['reading', 'coding']
}

with open('output.json', 'w') as file:
    json.dump(data, file, indent=4)

# Pretty print JSON
with open('output.json', 'w') as file:
    json.dump(data, file, indent=4, sort_keys=True)
JSON String Conversions
EXAMPLE 
import json

# Python to JSON string
data = {'name': 'Alice', 'age': 25}
json_string = json.dumps(data)
print(json_string)  # '{"name": "Alice", "age": 25}'

# JSON string to Python
json_string = '{"name": "Bob", "age": 30}'
data = json.loads(json_string)
print(data['name'])  # Bob
Real-World Example: Config File Handler
import json

class ConfigManager:
    def __init__(self, config_file):
        self.config_file = config_file
        self.config = self.load()
    
    def load(self):
        try:
            with open(self.config_file) as f:
                return json.load(f)
        except FileNotFoundError:
            return {}
    
    def save(self):
        with open(self.config_file, 'w') as f:
            json.dump(self.config, f, indent=4)
    
    def get(self, key, default=None):
        return self.config.get(key, default)
    
    def set(self, key, value):
        self.config[key] = value
        self.save()

# Usage
config = ConfigManager('config.json')
config.set('api_key', 'abc123')
print(config.get('api_key'))

pathlib - Modern Path Handling

Working with Paths
WHAT IT DOES

pathlib provides object-oriented filesystem paths. Works across platforms (Windows, Linux, Mac).

EXAMPLE 
from pathlib import Path

# Create path object
path = Path('data/files/document.txt')

# Path properties
print(path.name)        # document.txt
print(path.stem)        # document
print(path.suffix)      # .txt
print(path.parent)      # data/files

# Check if exists
print(path.exists())      # True/False
print(path.is_file())     # True/False
print(path.is_dir())      # True/False
File Operations with pathlib
EXAMPLE 
from pathlib import Path

path = Path('data.txt')

# Read file
content = path.read_text()
print(content)

# Write file
path.write_text("Hello, World!")

# List directory
directory = Path('.')
for file in directory.iterdir():
    print(file.name)

# Find all Python files
for py_file in Path('.').glob('*.py'):
    print(py_file)

# Recursive search
for py_file in Path('.').rglob('*.py'):
    print(py_file)
Creating Directories
EXAMPLE 
from pathlib import Path

# Create single directory
path = Path('new_folder')
path.mkdir(exist_ok=True)  # Won't error if exists

# Create nested directories
path = Path('data/logs/2024')
path.mkdir(parents=True, exist_ok=True)

# Join paths (cross-platform)
base = Path('data')
full_path = base / 'logs' / 'app.log'
print(full_path)  # data/logs/app.log

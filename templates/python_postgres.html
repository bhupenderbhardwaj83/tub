Setup & Connection

Installing psycopg2
WHAT IT DOES

psycopg2 is the most popular PostgreSQL adapter for Python. It allows you to connect and execute SQL queries.

EXAMPLE 
# Install psycopg2
# pip install psycopg2-binary

import psycopg2

# Connect to database
conn = psycopg2.connect(
    host="localhost",
    database="mydb",
    user="postgres",
    password="yourpassword",
    port=5432
)

# Create cursor
cursor = conn.cursor()

print("Connected to PostgreSQL!")
Using Context Manager
EXAMPLE 
import psycopg2
from psycopg2 import pool

# Context manager (auto-closes connection)
with psycopg2.connect(
    host="localhost",
    database="mydb",
    user="postgres",
    password="yourpassword"
) as conn:
    with conn.cursor() as cursor:
        cursor.execute("SELECT version();")
        version = cursor.fetchone()
        print(version)
Connection Pooling
EXAMPLE 
from psycopg2 import pool

# Create connection pool
connection_pool = pool.SimpleConnectionPool(
    1, 20,  # Min and max connections
    host="localhost",
    database="mydb",
    user="postgres",
    password="yourpassword"
)

# Get connection from pool
conn = connection_pool.getconn()

# Return connection to pool
connection_pool.putconn(conn)

# Close all connections
connection_pool.closeall()

CRUD Operations

CREATE - Insert Data
EXAMPLE 
import psycopg2

conn = psycopg2.connect(host="localhost", database="mydb", 
                        user="postgres", password="pass")
cursor = conn.cursor()

# Create table
cursor.execute("""
    CREATE TABLE IF NOT EXISTS users (
        id SERIAL PRIMARY KEY,
        username VARCHAR(50) UNIQUE NOT NULL,
        email VARCHAR(100) NOT NULL,
        created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
    )
""")

# Insert single row
cursor.execute(
    "INSERT INTO users (username, email) VALUES (%s, %s)",
    ("alice", "alice@email.com")
)

# Insert multiple rows
users = [
    ("bob", "bob@email.com"),
    ("charlie", "charlie@email.com")
]
cursor.executemany(
    "INSERT INTO users (username, email) VALUES (%s, %s)",
    users
)

conn.commit()  # Important!
cursor.close()
conn.close()
READ - Query Data
EXAMPLE 
import psycopg2

conn = psycopg2.connect(host="localhost", database="mydb",
                        user="postgres", password="pass")
cursor = conn.cursor()

# Fetch all rows
cursor.execute("SELECT * FROM users")
rows = cursor.fetchall()
for row in rows:
    print(row)

# Fetch one row
cursor.execute("SELECT * FROM users WHERE username = %s", ("alice",))
user = cursor.fetchone()
print(user)

# Fetch with column names
cursor.execute("SELECT * FROM users")
columns = [desc[0] for desc in cursor.description]
rows = cursor.fetchall()
for row in rows:
    user_dict = dict(zip(columns, row))
    print(user_dict)

cursor.close()
conn.close()
UPDATE & DELETE
EXAMPLE

Advanced Queries

Joins & Aggregations
EXAMPLE 
import psycopg2

conn = psycopg2.connect(host="localhost", database="mydb",
                        user="postgres", password="pass")
cursor = conn.cursor()

# INNER JOIN
cursor.execute("""
    SELECT users.username, posts.title, posts.created_at
    FROM users
    INNER JOIN posts ON users.id = posts.user_id
    WHERE users.username = %s
""", ("alice",))

results = cursor.fetchall()
for row in results:
    print(row)

# Aggregation with GROUP BY
cursor.execute("""
    SELECT users.username, COUNT(posts.id) as post_count
    FROM users
    LEFT JOIN posts ON users.id = posts.user_id
    GROUP BY users.username
    ORDER BY post_count DESC
""")

results = cursor.fetchall()
for username, count in results:
    print(f"{username}: {count} posts")

cursor.close()
conn.close()
Transactions
EXAMPLE 
import psycopg2

conn = psycopg2.connect(host="localhost", database="mydb",
                        user="postgres", password="pass")
cursor = conn.cursor()

try:
    # Begin transaction (automatic)
    cursor.execute(
        "INSERT INTO users (username, email) VALUES (%s, %s)",
        ("dave", "dave@email.com")
    )
    
    cursor.execute(
        "UPDATE accounts SET balance = balance - 100 WHERE user_id = %s",
        (1,)
    )
    
    # Commit transaction
    conn.commit()
    print("Transaction successful")
    
except Exception as e:
    # Rollback on error
    conn.rollback()
    print(f"Transaction failed: {e}")
    
finally:
    cursor.close()
    conn.close()
Real-World Example: Database Helper Class
import psycopg2
from psycopg2.extras import RealDictCursor

class Database:
    def __init__(self, config):
        self.config = config
        self.conn = None
    
    def connect(self):
        self.conn = psycopg2.connect(**self.config)
    
    def execute_query(self, query, params=None):
        with self.conn.cursor(cursor_factory=RealDictCursor) as cursor:
            cursor.execute(query, params)
            return cursor.fetchall()
    
    def execute_update(self, query, params=None):
        with self.conn.cursor() as cursor:
            cursor.execute(query, params)
            self.conn.commit()
            return cursor.rowcount
    
    def close(self):
        if self.conn:
            self.conn.close()

# Usage
config = {
    'host': 'localhost',
    'database': 'mydb',
    'user': 'postgres',
    'password': 'pass'
}

db = Database(config)
db.connect()
users = db.execute_query("SELECT * FROM users")
db.close()

pgAdmin - GUI Tool

Getting Started with pgAdmin
WHAT IT DOES

pgAdmin is the most popular PostgreSQL GUI tool. Manage databases, run queries, and visualize data visually.

Key Features:
• Visual database design and management
• Query tool with syntax highlighting
• Backup and restore databases
• User and permission management
• Real-time monitoring and statistics
• Import/Export data (CSV, JSON, etc.)
Common pgAdmin Tasks
1. Creating a Database:
Right-click "Databases" → Create → Database

2. Running Queries:
Right-click database → Query Tool → Write SQL → Execute (F5)

3. Creating Tables:
Expand database → Right-click "Tables" → Create → Table

4. Viewing Data:
Right-click table → View/Edit Data → All Rows

5. Import CSV:
Right-click table → Import/Export → Select file → Configure columns

6. Backup Database:
Right-click database → Backup → Choose format (Custom/Tar) → Execute
Useful SQL Queries in pgAdmin
EXAMPLE 
-- View all tables in current database
SELECT tablename FROM pg_tables
WHERE schemaname = 'public';

-- View table structure
SELECT column_name, data_type, is_nullable
FROM information_schema.columns
WHERE table_name = 'users';

-- Database size
SELECT pg_size_pretty(pg_database_size('mydb'));

-- Table row count
SELECT COUNT(*) FROM users;

-- Active connections
SELECT pid, usename, application_name, state
FROM pg_stat_activity;

Best Practices

Security & Performance
1. Use Parameterized Queries (Prevent SQL Injection): 
# BAD - SQL Injection risk
username = "alice'; DROP TABLE users; --"
cursor.execute(f"SELECT * FROM users WHERE username = '{username}'")

# GOOD - Parameterized query
cursor.execute("SELECT * FROM users WHERE username = %s", (username,))
2. Use Connection Pooling:
• Reuse connections instead of creating new ones
• Improves performance for web applications

3. Create Indexes: 
-- Create index for faster queries
CREATE INDEX idx_username ON users(username);

-- Composite index
CREATE INDEX idx_user_email ON users(username, email);
4. Use Transactions for Related Operations:
• Ensures data consistency
• All-or-nothing approach

5. Close Connections Properly:
• Always use context managers or close connections
• Prevents connection leaks

6. Handle Exceptions:
• Wrap database operations in try-except
• Rollback on errors
• Log errors for debugging
Environment Variables for Credentials
EXAMPLE 
import os

    host=os.getenv('DB_HOST'),
    database=os.getenv('DB_NAME'),

    password=os.getenv('DB_PASSWORD')
)

# pip install python-dotenv

load_dotenv()

    host=os.getenv('DB_HOST'),
